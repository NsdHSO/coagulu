import {
  ChangeDetectionStrategy,
  Component,
  inject,
  Input,
} from '@angular/core';
import { CommonModule } from '@angular/common';
import { TypeBadge, TypeSide } from '../../utils';
import { MatIconModule } from '@angular/material/icon';
import { GenerativeService } from '../../service';

export interface ActionBadge {
  side: TypeSide;
  action: () => unknown;
  icon?: string;
  text: string;
}
@Component({
  selector: 'ngx-form-badge',
  standalone: true,
  imports: [CommonModule, MatIconModule],
  templateUrl: './form-badge.component.html',
  styleUrls: ['./form-badge.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush,
})
export class FormBadgeComponent {
  protected readonly _generativeService = inject(GenerativeService);

  /**
   * Type of badge
   */
  @Input()
  typeBadge: TypeBadge = 'error';

  /**
   * Action
   */
  @Input()
  actionBadge: ActionBadge[] = [
    { side: 'middle', action: () => console.log('TEST'), text: 'Nothig' },
  ];
}

<div
  class="wrapper-badge rounded-xl border-solid border-1 drop-shadow-md text prose prose-zinc prose-lg border-2"
  [ngClass]="{
    'bg-green-300 border-green-300 text-green-700': typeBadge === 'success',
    'bg-blue-300 border-blue-300 text-blue-700 ': typeBadge === 'done',
    'bg-red-300 border-red-300 text-red-700 ': typeBadge === 'error',
    'bg-yellow-400 border-yellow-300 text-yellow-700 ': typeBadge === 'warning',
    'bg-blue-100 border-blue-300 text-blue-700 ': typeBadge === 'open',
    'bg-white border-gray-300': typeBadge === 'pfm'
  }"
>
  <ng-container *ngFor="let badge of actionBadge">
    <ng-container *ngIf="badge.side === 'left' && badge.icon">
      <mat-icon [svgIcon]="badge.icon" (click)="badge.action()" />
    </ng-container>
    <ng-container *ngIf="badge.side === 'middle' && !badge.icon">
      <div (click)="badge.action()">
        {{ badge.text }}
      </div>
    </ng-container>
    <ng-container *ngIf="badge.side === 'right' && badge.icon">
      <mat-icon [svgIcon]="badge.icon" (click)="badge.action()" />
    </ng-container>
  </ng-container>
</div>
