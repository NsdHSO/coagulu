// Helper function to generate random strings for ids
export function generateRandomId(): string {
  return Math.random().toString(36).substr(2, 40);
}
export interface Step {
  value: string;
  label: string;
  id: string;
}
export interface FormStepper {
  intro: InfoUser;
  infoUser: InfoUser;
  spendMoney: InfoUser;
  reserveBook: InfoUser;
  steps: Step[];
}

export interface ValuesClass {
  id?: string;
  graph?: Graph;
  reserve?: Reserve;
  details?: InfoUser;
  modals?: Modals;
  incoming?: number;
  spendMoney?: number;
  revenue?: number;
  categorySpendMoney?: CategorySpendMoney[];
  recommendations?: Recommendation[];
}

export interface InfoUser {
  check: boolean;
  id: string;
  info?: Info;
  icon?: string;
  values?: ValuesValue[] | ValuesClass | null;
}

export interface CategorySpendMoney {
  id: string;
  value: number;
  label: string;
  scope: Scope[];
}

export interface Scope {
  id: string;
  label: string;
  check: boolean;
}

export interface Graph {
  id: string;
  values: GraphValue[];
  howMannyMonth: number;
}

export interface GraphValue {
  id: string;
  label: string;
  start: number;
  end: number;
}

export interface Modals {
  setBookType: SetBookType;
  sumBook: SetBookType;
}

export interface SetBookType {
  id: string;
  value: number;
  labelCurrency: string;
}

export interface Recommendation {
  check: boolean;
  label: string;
}

export interface Reserve {
  id: string;
  currentReserve: SetBookType;
  recommendationReserve: SetBookType;
  howMannyMonth: number;
}

export interface ValuesValue {
  id: string;
  name: string;
  author: string;
  genre: string;
}

export interface Info {
  familyNumber: number;
  roleOfMember: RoleOfMember[];
  childrens: number;
}

export interface RoleOfMember {
  id: number;
  value: string;
  check: boolean;
}
